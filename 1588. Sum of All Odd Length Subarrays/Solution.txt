class Solution(object):
    def sumOddLengthSubarrays(self, arr):
        """
        :type arr: List[int]
        :rtype: int
        """
        result = sum(arr)
        length = len(arr)
        for j in range(3,length+1,2):
            for i in range(length):
                if (i+j) > length:
                    break
                result += sum(arr[i:i+j])
        return result
